# Route return traffic to Tailscale clients
ip rule add to 100.64.0.0/10 lookup 52 priority 95

# Forward Tailscale traffic through VPN tunnel
iptables -A FORWARD -i tailscale0 -o tun0 -j ACCEPT
iptables -A FORWARD -i tun0 -o tailscale0 -j ACCEPT

# Block any Tailscale traffic from bypassing VPN
iptables -A FORWARD -i tailscale0 -o eth0 -j DROP

# NAT outbound VPN traffic
iptables -t nat -A POSTROUTING -o tun0 -j MASQUERADE
